246       *   在子线程中执行 MQTT 连接操作​
247       */
248      private void mqtt_connect() {
249            new Thread(() -> {
250                try {
251                      if (!client.isConnected()) {
252                           client.connect(options);
253                           //   连接成功，通知 UI​
254                           handler.sendEmptyMessage(31);
255                      }
256                } catch (Exception e) {
257                      Log.e("MQTT  连接失败", e.getMessage());
258                      e.printStackTrace();
259                      //   连接失败，通知 UI​
260                      handler.sendEmptyMessage(30);
261                }
262            }).start();
263      }

265      /**
266       *   启动定时重连任务：每10秒检查一次连接状态，若断开则尝试重连​
267       */
268      private void start_reconnect() {
269            scheduler = Executors.newSingleThreadScheduledExecutor();
270            scheduler.scheduleAtFixedRate(() -> {
271                if (!client.isConnected()) {
272                      mqtt_connect();
273                }
274             }, 0, 10, TimeUnit.SECONDS); //   延迟0秒，每隔10秒执行一次​
275       }
