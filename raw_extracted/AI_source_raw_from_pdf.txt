120      /**
121       *   异步任务：调用 DashScope Qwen 模型，根据健康数据生成 AI 健康建议​
122       */
123      private class GenerateAIHealthReportTask extends AsyncTask<List<String[]>,
      Void, String> {
124
125            @Override
126            protected void onPreExecute() {
127                //   显示加载提示​
128                analysisTextView.setText("   正在分析健康数据，生成 AI 健康建议...");
129            }
130
131            @Override
132            protected String doInBackground(List<String[]>... params) {
133                List<String[]> data = params[0];
134
135                //   构建发送给大模型的提示词（Prompt）​
136                StringBuilder promptBuilder = new StringBuilder();
137                                 你是一位专业健康管理师。请根据以下用户健康监测历史数
                   promptBuilder.append("
      据，分析其整体健康状况，并给出具体、实用的健康建议。\n\n");
138           promptBuilder.append("数据格式：时间, 体温(℃), 心率(bpm), 血氧(%), 步
      数, 是否摔倒, 报警信息\n");
139
140                //   拼接每一条健康记录​
141                for (String[] row : data) {
142                     if (row.length >= 7) {
143                         promptBuilder.append(String.join(", ", row)).append("\n");
144                   }
145              }
146
147              // 指定 AI 回答的结构要求（使用 Unicode 图标替代数字编号）​
148              promptBuilder.append("\n请用中文分点回答，包括：\n");
149              promptBuilder.append("\u26AB 整体健康评估；\n"); // ⚫ 黑色圆
      点​
150              promptBuilder.append("\u26A0   异常情况总结；\n");               //   ⚠️ 警告三
      角​
151              promptBuilder.append("\uD83D\uDCD6   具体生活或医疗建议；\n");           // 📖 记
      事本​
152              promptBuilder.append("\uD83C\uDFE5   是否需要就医或进一步检查。\n"); //
      🏥 医院   ​
153
154              //   最终 Prompt 字符串​
155              String prompt = promptBuilder.toString();
156
157              //   构造符合 DashScope 兼容 OpenAI 格式的 JSON 请求体​
158              String jsonBody = "{"
159                       + "\"model\": \"qwen-plus\","
160                       + "\"messages\": [{\"role\": \"user\", \"content\": \"" +
      escapeJson(prompt) + "\"}],"
161                       + "\"max_tokens\": 600"
162                       + "}";
163
164              //   创建带超时设置的 OkHttpClient（防止网络请求卡死）​
165              OkHttpClient client = new OkHttpClient.Builder()
166                       .connectTimeout(10, java.util.concurrent.TimeUnit.SECONDS)
       //   连接超时​
167                       .readTimeout(25, java.util.concurrent.TimeUnit.SECONDS)
       //   读取超时​
168                       .writeTimeout(10, java.util.concurrent.TimeUnit.SECONDS)
       //   写入超时​
169                       .build();
170
171              //   设置请求头和请求体​
172              MediaType JSON = MediaType.get("application/json; charset=utf-8");
173              RequestBody body = RequestBody.create(jsonBody, JSON);
174              Request request = new Request.Builder()
175                       .url(BASE_URL)
176                       .addHeader("Authorization", "Bearer " + API_KEY)
177                       .addHeader("Content-Type", "application/json")
178                       .post(body)
179                       .build();
180
181              try (Response response = client.newCall(request).execute()) {
182                   if (response.isSuccessful() && response.body() != null) {
183                           //   成功收到响应，解析 JSON​
184                           String responseBody = response.body().string();
185                           Gson gson = new Gson();
186                           ChatResponse chatResponse = gson.fromJson(responseBody,
      ChatResponse.class);
187                           if (chatResponse.choices != null &&
      chatResponse.choices.length > 0) {
188                                //   返回 AI 生成的建议文本​
189                                return chatResponse.choices[0].message.content.trim();
190                           }
191                       } else {
192                           // HTTP 状态码非 2xx​
193                           return "AI 服务返回错误：" + response.code() + " - " +
      response.message();
194                       }
195                  } catch (IOException e) {
196                       e.printStackTrace();
197                       return " 网络请求失败：" + e.getMessage();
198                  }
199
200                  // 默认兜底返回​
201                  return "未能生成 AI 健康建议，请稍后重试。";
202            }
203

210            /**
211            * 对输入字符串进行简单 JSON 转义，防止破坏 JSON 结构​
212            * （例如引号、换行符等特殊字符）​
213            */
214            private String escapeJson(String input) {
215                  if (input == null) return "";
216                  return input.replace("\\", "\\\\")
217                           .replace("\"", "\\\"")
218                           .replace("\n", "\\n")
219                           .replace("\r", "\\r")
220                           .replace("\t", "\\t");
221            }
222      }
